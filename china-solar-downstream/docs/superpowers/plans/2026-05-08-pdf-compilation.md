## Task 2: Create master build script

**File:** `scripts/build_pdf.py`

- [ ] **Step 1: Write the compilation script**

The script must:
1. Define the ordered list of all 47 source files with their Part/Chapter labels
2. Read each markdown file and prepend a page-break div + part/chapter header where needed
3. Use `pandoc` to convert combined markdown → HTML (with `--toc`, `--toc-depth=3`, CSS link)
4. Inject cover page HTML before `<body>` content
5. Call `weasyprint` to render HTML → PDF
6. Print progress and final file size

```python
#!/usr/bin/env python3
"""
Build PhD-level PDF from all research documents.
Usage: python3 scripts/build_pdf.py
Output: docs/pdf/china-solar-downstream-2026.pdf
"""
import subprocess
import os
import sys
from pathlib import Path

BASE = Path(__file__).parent.parent

def build_combined_markdown():
    parts = []
    for label, rel_path, part_hdr, chap_hdr in MANIFEST:
        fp = BASE / rel_path
        if not fp.exists():
            print(f"  WARNING: missing {fp}", file=sys.stderr)
            continue
        content = fp.read_text(encoding="utf-8")
        section_parts = []
        if part_hdr:
            section_parts.append(f'\n\n<div class="part-header"><h1 class="part-title">{part_hdr}</h1></div>\n\n')
        if chap_hdr:
            section_parts.append(f'\n\n---\n\n## {chap_hdr}\n\n')
        section_parts.append(content)
        parts.append("".join(section_parts))
        print(f"  ✓ {label}")
    return "\n\n---\n\n".join(parts)


def main():
    OUT_DIR.mkdir(parents=True, exist_ok=True)
    print("Step 1: Assembling markdown...")
    combined_md = build_combined_markdown()

    combined_md_path = OUT_DIR / "_combined.md"
    combined_md_path.write_text(combined_md, encoding="utf-8")
    print(f"  Combined: {len(combined_md):,} chars, {len(combined_md.split()):,} words")

    print("Step 2: Converting markdown → HTML via pandoc...")
    html_path = OUT_DIR / "_report.html"
    result = subprocess.run([
        "pandoc",
        str(combined_md_path),
        "--from", "markdown+smart",
        "--to", "html5",
        "--standalone",
        "--toc",
        "--toc-depth=3",
        "--css", str(CSS),
        "--metadata", "title=China Solar Downstream: Business Model Evolution and Korea Applicability (2016–2026)",
        "--metadata", "lang=en",
        "--output", str(html_path),
    ], capture_output=True, text=True)
    if result.returncode != 0:
        print(f"  ERROR: {result.stderr}")
        sys.exit(1)
    print(f"  HTML: {html_path.stat().st_size / 1024:.0f} KB")

    print("Step 3: Injecting cover page...")
    html_content = html_path.read_text(encoding="utf-8")
    # Insert cover after <body>
    html_content = html_content.replace("<body>", f"<body>\n{COVER_HTML}", 1)
    html_path.write_text(html_content, encoding="utf-8")

    print("Step 4: Rendering HTML → PDF via weasyprint...")
    result = subprocess.run([
        "weasyprint",
        str(html_path),
        str(OUT_PDF),
    ], capture_output=True, text=True)
    if result.returncode != 0:
        print(f"  ERROR: {result.stderr[:500]}")
        sys.exit(1)

    size_mb = OUT_PDF.stat().st_size / (1024 * 1024)
    print(f"\n✅ PDF generated: {OUT_PDF}")
    print(f"   Size: {size_mb:.1f} MB")


if __name__ == "__main__":
    main()
```
